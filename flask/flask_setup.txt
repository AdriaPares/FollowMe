# Seting up Flask Server

sudo apt-get update
sudo apt install python3-pip -y

# Install python libraries. flask-cassandra also installs cassandra driver, which takes a while

pip3 install flask --user
pip3 install flask-cassandra --user

# Test simple_flask_cassandra.py (make sure ports are open, that Cassandra has something in those tables,
# and that you have simple_flask_cassandra.py in the Flask instance)
# Make sure to configure the Cassandra cluster nodes in simple_flask_cassandra.py, as well as the tables we are
# querying

export FLASK_APP=simple_flask_cassandra.py
flask run --host=0.0.0.0
